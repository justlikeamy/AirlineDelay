fixing airport data¶

Some of the airports were listed using 5 digit numbers instead of 3 letter code. I used code by Scott Cole as a template for this section (see the link below). First fornatted all the codes, then make sure that they are strings. For all the codes that aren't 3 letters, look up their corresponding 5 digit code and replace. https://www.kaggle.com/srcole/fix-inconsistent-airport-codes

aircode1 = pd.read_csv('AirportCodes.csv', encoding='latin-1') aircode2 = pd.read_csv('FiveDigitAirportID.csv', encoding='latin-1')

aircode1 = aircode1.reset_index() aircode2 = aircode2.reset_index() aircodes = pd.merge(aircode1,aircode2,on='Description') aircode_dict = dict(zip(aircodes['ID'].astype(str),aircodes['Code']))

flights['ORIGIN_AIRPORT'] = flights['ORIGIN_AIRPORT'].values.astype(str) flights['DESTINATION_AIRPORT'] = flights['DESTINATION_AIRPORT'].values.astype(str)

for i in range(len(flights)): print(i) if len(flights['ORIGIN_AIRPORT'][i]) != 3: to_replace = flights['ORIGIN_AIRPORT'][i] value = aircode_dict[flights['ORIGIN_AIRPORT'][i]] flights = flights.replace(to_replace, value)

if len(flights['DESTINATION_AIRPORT'][i]) != 3:
    to_replace = flights['DESTINATION_AIRPORT'][i]
    value = aircode_dict[flights['DESTINATION_AIRPORT'][i]]
    flights = flights.replace(to_replace, value)

def late_or_cancelled(x):
    #if x['CANCELLED'] == 1 or x['total delay'] > 15:
    if x['CANCELLED'] == 1 or x['ARRIVAL_DELAY'] >= 60 or x['DEPARTURE_DELAY'] >= 60:
        return 1
    else:
        return 0

flights['late or cancelled'] = flights.apply(late_or_cancelled,axis = 1)
